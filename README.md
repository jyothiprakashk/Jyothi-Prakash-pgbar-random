## Install

```bash
npm install --save @jyothi/react-pg-bar
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from '@jyothi/react-pg-bar'
import '@jyothi/react-pg-bar/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
